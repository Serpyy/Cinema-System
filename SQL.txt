CREATE TABLE movie
(	movie_ID		VARCHAR(4) 	NOT NULL primary key,
	movie_title		VARCHAR(30) NOT NULL,
	duration		INTEGER	NOT NULL,
	release_date	DATE		NOT NULL,
	synopsis		VARCHAR(500)NOT NULL,
	director		VARCHAR(30)	NOT NULL,
	distributor		VARCHAR(30)	NOT NULL,
	ageRating		VARCHAR(3)		NOT NULL,
	poster			BLOB
);

CREATE TABLE actor
(	actor_ID		VARCHAR(4) 	NOT NULL primary key,
	actor_name		VARCHAR(30) NOT NULL
);

CREATE TABLE CASTS(
      movie_ID		VARCHAR(4) 	NOT NULL,
	actor_ID		VARCHAR(4) 	NOT NULL,
primary key(movie_ID,actor_ID),
FOREIGN KEY (movie_ID) REFERENCES MOVIE(movie_id),
FOREIGN KEY (actor_ID) REFERENCES ACTOR(actor_id)	
);

CREATE Table Hall (
hall_no varchar(3) Not NULL,
row  int NOT NULL,
columns int NOT NULL,
Primary Key (Hall_no)
);	

CREATE Table Schedule (
schedule_no varchar(7) Not NULL,
movie_ID varchar(4) Not NULL,
hall_no varchar(3) Not NULL,
schedule_time TIME NOT NULL,
schedule_date DATE NOT NULL,
PRIMARY KEY (schedule_no),
FOREIGN KEY (movie_ID) REFERENCES movie(movie_ID),
FOREIGN KEY (hall_no) REFERENCES hall(hall_no)
);	

CREATE TABLE Customer (
   cust_ID              VARCHAR(5) not null primary key,
   cust_name           	VARCHAR(50) not null,     
   ic_no                VARCHAR(16) not null,
   email           	VARCHAR(50) not null unique,
   cust_password        VARCHAR(16) not null,
   phone_no         	BIGINT not null,
   security_question    VARCHAR(255) not null,  
   answer 		VARCHAR(50) not null,
   point                INTEGER
);

CREATE TABLE Card (
   card_no		VARCHAR(16) not null primary key,
   cust_ID		VARCHAR(5) not null,
   cvv			VARCHAR(3) not null,
   cardholder_name	VARCHAR(50) not null,
   expiry_date		DATE not null,
   FOREIGN KEY (cust_ID) REFERENCES Customer(cust_ID)
);

CREATE TABLE Payment(
	payment_no VARCHAR(6) not null primary key,
	card_no VARCHAR(16) NOT NULL,
	amount NUMERIC(6,2) NOT NULL,
	payment_time TIME NOT NULL,
	payment_date DATE NOT NULL,
	child_qty INTEGER NOT NULL,
	adult_qty INTEGER NOT NULL,
	foreign key (card_no) references Card(card_no)
);

CREATE TABLE Ticket(
	ticket_no VARCHAR(10) not null primary key,
	payment_no VARCHAR(6) NOT NULL,
	schedule_no VARCHAR(7) NOT NULL,
	price NUMERIC(4,2) NOT NULL,
	category VARCHAR(5) NOT NULL,
	seat_no VARCHAR(3) NOT NULL,
	foreign key (payment_no) references Payment(payment_no),
	foreign key (schedule_no) references Schedule(schedule_no)
);

CREATE TABLE Staff (
	staff_ID varchar(4) NOT NULL,
	staff_name varchar(255) NOT NULL,
	staff_email varchar(255) NOT NULL,
	password varchar(255) NOT NULL,
	role varchar(50) NOT NULL,
	staff_phone_no varchar(12) NOT NULL,
	security_question varchar(255),
	answer varchar(255),
    PRIMARY KEY (staff_ID)
);

CREATE TABLE Review (
	review_no varchar(6) NOT NULL,
	cust_ID varchar(5) NOT NULL,
	comment varchar(2000),
	rating int NOT NULL CHECK(rating BETWEEN 1 AND 5),
	response varchar(2000),
    PRIMARY KEY (review_no),
    FOREIGN KEY (cust_ID) REFERENCES Customer(cust_ID)
);

CREATE TABLE refund(
	refund_no	VARCHAR(8) NOT NULL,
	payment_no      VARCHAR(6) NOT NULL,
	staff_id	VARCHAR(4)NOT NULL,
	refund_date	Date NOT NULL,
	reason		VARCHAR(100) NOT NULL ,
	status		char(1) NOT NULL,
PRIMARY KEY(refund_no),
FOREIGN KEY(payment_no) REFERENCES payment(payment_no),
FOREIGN KEY(staff_ID) REFERENCES staff(staff_ID),
constraint chk_status check (UPPER(status) in ('A','U','P'))
);

CREATE TABLE voucher(
	voucher_no	VARCHAR(8) NOT NULL,
	cust_ID		VARCHAR(5) NOT NULL,
	dis_amount	Int NOT NULL,
PRIMARY KEY(voucher_no),
FOREIGN KEY(cust_ID) REFERENCES customer(cust_ID)
);
